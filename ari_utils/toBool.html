bool toBool(val) {
  if (val is bool) {
    return val;
  }
  if (val == null) {
    return false;
  }
  if (val is num) {
    return val != 0;
  }
  // For strings|iterables
  // <editor-fold desc="Is not empty">
  try {
    return val.isNotEmpty;
  } on NoSuchMethodError {
    // Pass
  }
  // </editor-fold>
  return true;
}
